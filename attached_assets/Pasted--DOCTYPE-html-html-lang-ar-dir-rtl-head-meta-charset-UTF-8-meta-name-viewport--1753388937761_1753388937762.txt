  <script>
    const appID = parseInt("YOUR_APP_ID");
    const serverSecret = "YOUR_SERVER_SECRET";
    const zg = new ZegoExpressEngine(appID, serverSecret);
    const viewerVideo = document.getElementById("viewer");

    // بثوث حقيقية فقط يتم عرضها
    const liveStreams = [
      {
        username: "أحمد",
        image: "https://via.placeholder.com/240x135.png?text=Live+1",
        roomID: "room_ahmed",
        streamID: "stream_ahmed",
        isLive: true
      },
      {
        username: "نور",
        image: "https://via.placeholder.com/240x135.png?text=Live+2",
        roomID: "room_noor",
        streamID: "stream_noor",
        isLive: true
      }
    ];

    async function checkLiveStatus(roomID, streamID) {
      try {
        const userID = "check_" + Date.now();
        const userName = "مراقب";
        await zg.loginRoom(roomID, { userID, userName }, { token: "" });
        const streamList = await zg.getRemoteStreamList();
        await zg.logoutRoom(roomID);
        return streamList.some(s => s.streamID === streamID);
      } catch (e) {
        return false;
      }
    }

    async function playStream(roomID, streamID) {
      const userID = "viewer_" + Date.now();
      const userName = "مشاهد";
      await zg.loginRoom(roomID, { userID, userName }, { token: "" });
      const remoteStream = await zg.startPlayingStream(streamID);
      viewerVideo.srcObject = remoteStream;
      viewerVideo.style.display = "block";
    }

    async function renderStreams() {
      const list = document.getElementById("streamList");
      for (const stream of liveStreams) {
        const isLive = await checkLiveStatus(stream.roomID, stream.streamID);
        if (!isLive) continue;

        const card = document.createElement("div");
        card.className = "stream-card";

        const badge = document.createElement("div");
        badge.className = "live-badge";
        badge.textContent = "🔴 مباشر";
        card.appendChild(badge);

        const img = document.createElement("img");
        img.src = stream.image;
        card.appendChild(img);

        const name = document.createElement("p");
        name.textContent = stream.username;
        card.appendChild(name);

        const btn = document.createElement("button");
        btn.textContent = "مشاهدة البث";
        btn.onclick = () => playStream(stream.roomID, stream.streamID);
        card.appendChild(btn);

        list.appendChild(card);
      }
    }

    renderStreams();
  </script>
